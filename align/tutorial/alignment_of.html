alignment_of

This type trait can be used to query the alignment requirement of a type at compile time. It is useful in static assertions to validate type alignment. 

#include <boost/align/alignment_of.hpp>

template<class T>
class utility {
  static_assert(boost::alignment::
    alignment_of<T>::value == 16, "");
public:
  utility()
    : object() {
  }
private:
  T object;
};

struct alignas(16) type {
  float data[4];
};

int main()
{
  utility<type> u;
  (void)u;
}
