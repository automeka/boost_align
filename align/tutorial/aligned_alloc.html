aligned_alloc and aligned_free

Consider these functions alignment enabled versions of std::malloc(std::size_t) and std::free(void*) or ::operator new(std::size_t, const std::no_throw_t&) and ::operator delete(void*). The aligned allocation function allocates space with the specified size and alignment. The aligned deallocation function can then deallocate this space. 

#include <boost/align/aligned_alloc.hpp>

void use(void* ptr, std::size_t size)
{
  (void)ptr;
  (void)size;
}

int main()
{
  void* p = boost::alignment::aligned_alloc(16, 100);
  if (p) {
    use(p, 100);
    boost::alignment::aligned_free(p);
  }
}
